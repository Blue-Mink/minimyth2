#
# To prepare u-boot script, run:
# mkimage -A arm64 -T script -O linux -d h6-boot-debug.txt h6-boot-debug.scr
# to debug DRM add: drm.debug=0xe

part uuid ${devtype} ${devnum}:2 uuid

setenv bootargs root=PARTUUID=${uuid} ro rootwait earlycon console=ttyS0,115200n8 debug MM_DEBUG="yes" debug
setenv fdtfile h6_dtb
setenv macaddr 42 5b 82 e4 85 b4

if load ${devtype} ${devnum}:${bootpart} ${kernel_addr_r} /Image; then
  if load ${devtype} ${devnum}:${bootpart} ${fdt_addr_r} /${fdtfile}; then
    fdt addr ${fdt_addr_r}
    fdt resize
    booti ${kernel_addr_r} - ${fdt_addr_r};
  else
    booti ${kernel_addr_r} - ${fdtcontroladdr};
  fi;
fi
